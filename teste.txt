};

// Funcao para armazenar uma forma geometrica na lista de formas
// Armazena sempre no inicio da lista
void pushForma(int tipo){
    forma f;
    f.tipo = tipo;
    formas.push_front(f);
}

// Funcao para armazenar um vertice na forma do inicio da lista de formas geometricas
// Armazena sempre no inicio da lista
void pushVertice(int x, int y){
    vertice v;
    v.x = x;
    v.y = y;
    formas.front().v.push_front(v);
}


// Função para armazenar uma circunferência na lista de formas geométricas
void pushCircunferencia(int x_c, int y_c, int r_c) {
    pushForma(CIR);
    pushVertice(x_c, y_c);
    pushVertice(r_c, 0); // armazenando o raio como um segundo vértice
}

void bresenhamLine(int x_1, int y_1, int x_2, int y_2){
    bool declive, simetrico;
    int dx = x_2 - x_1;
    int dy = y_2 - y_1;
   // int dx_ = dx, dy_ = dy, x_1_ = x_1, x_2_ = x_2;
    int xi , yi;
    int x_, y_;
    
    //Redu��o de octante
    declive = false;
    simetrico = false;
    if ((dx * dy) < 0) {
	    y_1 = (-1) * y_1;
	    y_2 = (-1) * y_2;
	    dy = (-1) * dy;
	    simetrico = true;
	}
	
	if (abs(dx) < abs(dy)) {
	    std::swap(x_1, y_1);
	    std::swap(x_2, y_2);
	    std::swap(dx, dy);
	    declive = true;
	}
	
	if (x_1 > x_2) {
	    std::swap(x_1, x_2);
	    std::swap(y_1, y_2);
	    dx = (-1) * dx;
	    dy = (-1) * dy;
	}
    
    xi = x_1;
    yi = y_1;
    //Algoritmo de Bresenham
    int delta = (2 * dy) - dx;
    int incE = 2 * dy;
    int incNE = 2 * (dy - dx);
    
   // drawPixel(xi, yi);
    while(xi < x_2){
        //incE
        if(delta <= 0){
            delta += incE;
        }else{ //incNE
            delta += incNE;
            yi += 1;
        }
        xi += 1;
        
        //Transforma��o para o octante original
        x_ = xi;
        y_ = yi;
        if(declive == true){
            x_ = yi;
            y_ = xi;
        }

        if(simetrico == true){
            y_ = (-1) * y_;
        }
        //pontos = pushPonto(x_, y_);
        drawPixel(x_, y_);
        //pushVertice(x_,y_);
    }
}
